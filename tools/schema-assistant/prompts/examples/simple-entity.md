### Example: Simple Entity (Resource)

**Request**: Create a Resource entity for booking rooms with name, capacity, and location.

**Response**:

A simple entity with basic fields. Resources are public-readable and authenticated-writable.

-- [DDL] Table creation
CREATE TABLE public.resources (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at TIMESTAMPTZ,
  display_name TEXT NOT NULL,
  capacity INT,
  building TEXT
);

ALTER TABLE public.resources ADD CONSTRAINT resources_pkey PRIMARY KEY (id);

-- [TRIGGERS] Timestamp triggers
CREATE TRIGGER set_created_at_trigger
  BEFORE INSERT ON public.resources
  FOR EACH ROW EXECUTE FUNCTION public.set_created_at();

CREATE TRIGGER set_updated_at_trigger
  BEFORE INSERT OR UPDATE ON public.resources
  FOR EACH ROW EXECUTE FUNCTION public.set_updated_at();

-- [GRANTS] Permission grants
GRANT SELECT, INSERT, UPDATE, DELETE ON TABLE public.resources TO web_anon, authenticated;
GRANT USAGE ON SEQUENCE public.resources_id_seq TO web_anon, authenticated;

-- [RLS] Row level security policies
ALTER TABLE public.resources ENABLE ROW LEVEL SECURITY;

CREATE POLICY "resources: read permission" ON public.resources
  FOR SELECT TO PUBLIC USING (has_permission('resources', 'read'));

CREATE POLICY "resources: create permission" ON public.resources
  FOR INSERT TO PUBLIC WITH CHECK (has_permission('resources', 'create'));

CREATE POLICY "resources: update permission" ON public.resources
  FOR UPDATE TO PUBLIC
  USING (has_permission('resources', 'update'))
  WITH CHECK (has_permission('resources', 'update'));

CREATE POLICY "resources: delete permission" ON public.resources
  FOR DELETE TO PUBLIC USING (has_permission('resources', 'delete'));

INSERT INTO metadata.permission_roles (permission_id, role_id)
